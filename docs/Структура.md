.
├── server-setup.sh         # Главный скрипт. Точка входа, отвечает за запуск меню и роутинг к модулям.
|
├── config/
│   └── server.conf        # Конфигурация. Все постоянные настройки и переменные окружения.
|
├── install.sh              # Скрипт для первичной установки Server Setup на сервер.
|
├── modules/                # Основные модули системы. Здесь живут крупные блоки функционала.
│   ├── core/               # Модули ядра: общие хелперы, генератор меню, обновление, сканер состояния.
│   │   ├── common.sh       # Общие функции: цвета, логирование, UI-хелперы, run_cmd, ввод/вывод.
│   │   ├── dependencies.sh # Хелпер для установки системных зависимостей (curl, jq, etc.).
│   │   ├── menu_generator.sh # Сканирует манифесты @menu.manifest и строит все меню в системе.
│   │   └── state_scanner.sh # Определяет наличие окружения и другие системные статусы.
│   │
│   ├── security/           # Модули для усиления безопасности сервера.
│   │   ├── system_update.sh # Обновление системы.
│   │   ├── mirror_check.sh # Управление зеркалами APT.
│   │   ├── ssh_setup.sh    # Настройка SSH (в разработке).
│   │   ├── firewall.sh     # Управление фаерволом (в разработке).
│   │   └── fail2ban.sh    # Управление Fail2Ban (в разработке).
│   │
│   └── ui/                 # Модули, связанные с отрисовкой пользовательского интерфейса.
│       ├── dashboard.sh    # Отрисовка главного дашборда.
│       └── widget_manager.sh # Управление виджетами дашборда.
|
└── plugins/                # Плагины для расширения функционала.
    ├── dashboard_widgets/  # Виджеты для главного дашборда.
    │   ├── 01_crypto_price.sh # Пример: курс криптовалюты.
    │   └── 02_load_short.sh   # Пример: краткий обзор системы.
    │
    └── commands/           # Команды, выполняемые на сервере.
        ├── security/       # Команды безопасности.
        │   └── 01_get_security_status.sh # Пример: получить статус безопасности.
        │
        └── system/         # Системные команды.
            └── 01_update_system.sh       # Пример: обновить систему.
